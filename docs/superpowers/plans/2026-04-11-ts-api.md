# ts-api Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the ts-api REST API module serving pre-aggregated metrics and per-request detail to the TokenScope frontend.

**Architecture:** Extend `StorageBackend` trait with query methods and implement them for DuckDB. Create a new `ts-api` crate with Axum handlers that convert HTTP params to storage queries and wrap results in a unified `ApiResponse<T>` envelope. The API server runs as a `tokio::spawn` task in `main.rs`.

**Tech Stack:** Rust, Axum 0.8, tower-http (CORS), DuckDB, serde, tokio

/// Build a WHERE clause fragment for dimension filtering.
/// When a dimension filter is empty, match the wildcard '*' row.
/// When specific values are given, match those exact values.
fn build_dimension_where(filter: &DimensionFilter) -> String {
    let mut clauses = Vec::new();

    if filter.providers.is_empty() {
        clauses.push("provider = '*'".to_string());
    } else {
        let vals: Vec<String> = filter.providers.iter().map(|v| format!("'{v}'")).collect();
        clauses.push(format!("provider IN ({})", vals.join(",")));
    }

    if filter.models.is_empty() {
        clauses.push("model = '*'".to_string());
    } else {
        let vals: Vec<String> = filter.models.iter().map(|v| format!("'{v}'")).collect();
        clauses.push(format!("model IN ({})", vals.join(",")));
    }

    if filter.server_ips.is_empty() {
        clauses.push("server_ip = '*'".to_string());
    } else {
        let vals: Vec<String> = filter.server_ips.iter().map(|v| format!("'{v}'")).collect();
        clauses.push(format!("server_ip IN ({})", vals.join(",")));
    }

    clauses.join(" AND ")
}

/// Convert a DimensionFilter for use with group_by queries.
/// When grouping by a dimension, we need to match specific (non-wildcard) rows
/// for that dimension, while keeping wildcard matching for other dimensions.
fn build_dimension_where_for_group(filter: &DimensionFilter, group_by: &str) -> String {
    let mut clauses = Vec::new();

    match group_by {
        "provider" => {
            // Match non-wildcard providers, wildcard model and server_ip
            if filter.providers.is_empty() {
                clauses.push("provider != '*'".to_string());
            } else {
                let vals: Vec<String> = filter.providers.iter().map(|v| format!("'{v}'")).collect();
                clauses.push(format!("provider IN ({})", vals.join(",")));
            }
            clauses.push("model = '*'".to_string());
            clauses.push("server_ip = '*'".to_string());
        }
        "model" => {
            // Match non-wildcard models (per-model rows have provider set too)
            clauses.push("provider != '*'".to_string());
            if filter.models.is_empty() {
                clauses.push("model != '*'".to_string());
            } else {
                let vals: Vec<String> = filter.models.iter().map(|v| format!("'{v}'")).collect();
                clauses.push(format!("model IN ({})", vals.join(",")));
            }
            clauses.push("server_ip = '*'".to_string());
        }
        _ => {}
    }

    clauses.join(" AND ")
}

    async fn query_metrics_timeseries(&self, query: &MetricsTimeseriesQuery) -> Result<Vec<MetricsTimeseriesRow>> {
        // Validate fields
        for field in &query.fields {
            if !VALID_METRIC_FIELDS.contains(&field.as_str()) {
                return Err(AppError::Storage(format!("invalid metric field: {field}")));
            }
        }

        let conn = self.conn.clone();
        let query = query.clone();
        tokio::task::spawn_blocking(move || {
            let conn = conn.lock().map_err(|e| {
                AppError::Storage(format!("failed to lock connection: {e}"))
            })?;

            let (sql, has_group) = if let Some(ref group_by) = query.group_by {
                // For group_by queries, we need to SUM count/total fields and
                // do weighted average for avg fields across the grouped dimension.
                let dim_where = build_dimension_where_for_group(&query.filter, group_by);

                // Build SELECT columns: for count/total fields use SUM,
                // for avg/percentile fields use weighted average by request_count.
                let field_selects: Vec<String> = query.fields.iter().map(|f| {
                    if f.contains("count") || f.starts_with("total_") || f == "concurrency_max" {
                        format!("SUM({f})")
                    } else if f == "concurrency_avg" || f.contains("avg") {
                        // Weighted average: SUM(field * request_count) / SUM(request_count)
                        format!("CASE WHEN SUM(request_count) > 0 THEN SUM({f} * request_count) / SUM(request_count) ELSE NULL END")
                    } else {
                        // Percentile fields: weighted average is an approximation
                        format!("CASE WHEN SUM(request_count) > 0 THEN SUM({f} * request_count) / SUM(request_count) ELSE NULL END")
                    }
                }).collect();

                let sql = format!(
                    "SELECT epoch(timestamp), {group_by}, {} \
                     FROM llm_metrics \
                     WHERE {dim_where} \
                       AND granularity = ? \
                       AND timestamp >= ? AND timestamp < ? \
                     GROUP BY timestamp, {group_by} \
                     ORDER BY timestamp, {group_by}",
                    field_selects.join(", "),
                );
                (sql, true)
            } else {
                let dim_where = build_dimension_where(&query.filter);
                let field_selects: Vec<String> = query.fields.iter().map(|f| f.to_string()).collect();

                let sql = format!(
                    "SELECT epoch(timestamp), {} \
                     FROM llm_metrics \
                     WHERE {dim_where} \
                       AND granularity = ? \
                       AND timestamp >= ? AND timestamp < ? \
                     ORDER BY timestamp",
                    field_selects.join(", "),
                );
                (sql, false)
            };

            let start_ts = us_to_timestamp(query.time_range.start_us);
            let end_ts = us_to_timestamp(query.time_range.end_us);

            let mut stmt = conn
                .prepare(&sql)
                .map_err(|e| AppError::Storage(format!("query error: {e}")))?;

            let field_count = query.fields.len();
            let rows = stmt
                .query_map(
                    duckdb::params![query.granularity, start_ts, end_ts],
                    |row| {
                        let timestamp: i64 = row.get(0)?;
                        let (group, field_offset) = if has_group {
                            (Some(row.get::<_, String>(1)?), 2)
                        } else {
                            (None, 1)
                        };
                        let mut values = Vec::with_capacity(field_count);
                        for i in 0..field_count {
                            values.push(row.get::<_, Option<f64>>(field_offset + i)?);
                        }
                        Ok(MetricsTimeseriesRow {
                            timestamp,
                            group,
                            values,
                        })
                    },
                )
                .map_err(|e| AppError::Storage(format!("query error: {e}")))?;

            let mut result = Vec::new();
            for row in rows {
                result.push(row.map_err(|e| AppError::Storage(format!("row error: {e}")))?);
            }
            Ok(result)
        })
        .await
        .map_err(|e| AppError::Storage(format!("spawn_blocking failed: {e}")))?

    async fn query_calls(&self, query: &CallsQuery) -> Result<CallsPage> {
        let sort_by = if VALID_CALL_SORT_FIELDS.contains(&query.sort_by.as_str()) {
            query.sort_by.clone()
        } else {
            "request_time".to_string()
        };
        let sort_order = if query.sort_order == "asc" { "ASC" } else { "DESC" };

        let conn = self.conn.clone();
        let query = query.clone();
        tokio::task::spawn_blocking(move || {
            let conn = conn.lock().map_err(|e| {
                AppError::Storage(format!("failed to lock connection: {e}"))
            })?;

            let start_ts = us_to_timestamp(query.time_range.start_us);
            let end_ts = us_to_timestamp(query.time_range.end_us);

            // Build WHERE clause
            let mut where_parts = vec![
                "request_time >= ?".to_string(),
                "request_time < ?".to_string(),
            ];
            if !query.filter.providers.is_empty() {
                let vals: Vec<String> = query.filter.providers.iter().map(|v| format!("'{v}'")).collect();
                where_parts.push(format!("provider IN ({})", vals.join(",")));
            }
            if !query.filter.models.is_empty() {
                let vals: Vec<String> = query.filter.models.iter().map(|v| format!("'{v}'")).collect();
                where_parts.push(format!("model IN ({})", vals.join(",")));
            }
            if !query.filter.server_ips.is_empty() {
                let vals: Vec<String> = query.filter.server_ips.iter().map(|v| format!("'{v}'")).collect();
                where_parts.push(format!("server_ip IN ({})", vals.join(",")));
            }
            if !query.status_codes.is_empty() {
                let vals: Vec<String> = query.status_codes.iter().map(|v| v.to_string()).collect();
                where_parts.push(format!("status_code IN ({})", vals.join(",")));
            }
            if !query.finish_reasons.is_empty() {
                let vals: Vec<String> = query.finish_reasons.iter().map(|v| format!("'{v}'")).collect();
                where_parts.push(format!("finish_reason IN ({})", vals.join(",")));
            }

            let where_clause = where_parts.join(" AND ");
            let offset = (query.page.saturating_sub(1)) * query.page_size;

            // Count query
            let count_sql = format!("SELECT COUNT(*) FROM llm_calls WHERE {where_clause}");
            let mut stmt = conn
                .prepare(&count_sql)
                .map_err(|e| AppError::Storage(format!("query error: {e}")))?;
            let total: u64 = stmt
                .query_row(duckdb::params![start_ts, end_ts], |row| row.get(0))
                .map_err(|e| AppError::Storage(format!("query error: {e}")))?;

            // Data query
            let data_sql = format!(
                "SELECT id, epoch(request_time), provider, model, status_code, is_stream, \
                        finish_reason, ttfb_ms, e2e_latency_ms, input_tokens, output_tokens \
                 FROM llm_calls \
                 WHERE {where_clause} \
                 ORDER BY {sort_by} {sort_order} \
                 LIMIT ? OFFSET ?",
            );

                .prepare(&data_sql)
                .map_err(|e| AppError::Storage(format!("query error: {e}")))?;
            let rows = stmt
                .query_map(
                    duckdb::params![start_ts, end_ts, query.page_size, offset],
                    |row| {
                        Ok(CallListItem {
                            id: row.get(0)?,
                            request_time: row.get(1)?,
                            provider: row.get(2)?,
                            model: row.get(3)?,
                            status_code: row.get(4)?,

                            e2e_latency_ms: row.get(8)?,
                            input_tokens: row.get(9)?,
                            output_tokens: row.get(10)?,
                        })

            for row in rows {

            }

use axum::http::StatusCode;
use axum::response::{IntoResponse, Response};
use serde::Serialize;

/// Unified API response envelope.
#[derive(Debug, Serialize)]
pub struct ApiResponse<T: Serialize> {
    pub code: i32,
    pub message: String,
    pub data: T,
}

impl<T: Serialize> ApiResponse<T> {
    pub fn ok(data: T) -> Self {
        Self {
            code: 0,
            message: "ok".to_string(),
            data,
        }
    }
}

impl<T: Serialize> IntoResponse for ApiResponse<T> {
    fn into_response(self) -> Response {
        let body = serde_json::to_string(&self).unwrap_or_else(|e| {
            format!(r#"{{"code":5001,"message":"serialization error: {e}","data":{{}}}}"#)
        });
        (
            StatusCode::OK,
            [("content-type", "application/json")],
            body,
        )
            .into_response()
    }
}

/// API error type — converts to the unified response envelope with `data: {}`.
#[derive(Debug)]
pub enum ApiError {
    /// 400 — invalid or missing parameter.
    InvalidParam(String),
    /// 404 — record not found.
    NotFound(String),
    /// 500 — internal server error.
    Internal(String),
}

/// Empty object for error responses (data: {}).
#[derive(Serialize)]
struct EmptyObject {}

impl IntoResponse for ApiError {
    fn into_response(self) -> Response {
        let (status, code, message) = match self {
            ApiError::InvalidParam(msg) => (StatusCode::BAD_REQUEST, 1001, msg),
            ApiError::NotFound(msg) => (StatusCode::NOT_FOUND, 2001, msg),
            ApiError::Internal(msg) => (StatusCode::INTERNAL_SERVER_ERROR, 5001, msg),
        };

        let body = ApiResponse {
            code,
            message,
            data: EmptyObject {},
        };

        let json = serde_json::to_string(&body).unwrap_or_else(|e| {
            format!(r#"{{"code":5001,"message":"serialization error: {e}","data":{{}}}}"#)
        });

        (status, [("content-type", "application/json")], json).into_response()
    }
}

impl From<ts_common::error::AppError> for ApiError {
    fn from(err: ts_common::error::AppError) -> Self {
        ApiError::Internal(err.to_string())
    }
}

const VALID_GRANULARITIES: &[&str] = &["10s", "1m", "5m", "1h"];

// --- Timeseries ---

#[derive(Serialize)]
struct TimeseriesSeries {
    name: String,
    group: Option<String>,
    values: Vec<Option<f64>>,
}

#[derive(Serialize)]
struct TimeseriesData {
    timestamps: Vec<i64>,
    series: Vec<TimeseriesSeries>,

pub async fn timeseries(
    State(storage): State<Arc<dyn StorageBackend>>,
    Query(params): Query<TimeseriesParams>,
) -> Result<impl IntoResponse, ApiError> {
    if !VALID_GRANULARITIES.contains(&params.granularity.as_str()) {
        return Err(ApiError::InvalidParam(format!(
            "granularity must be one of: {}",
            VALID_GRANULARITIES.join(", ")
        )));
    }

    let fields = parse_csv(&Some(params.fields.clone()));
    if fields.is_empty() {
        return Err(ApiError::InvalidParam("fields is required".to_string()));
    }

    if let Some(ref gb) = params.group_by {
        if gb != "provider" && gb != "model" {
            return Err(ApiError::InvalidParam(
                "group_by must be 'provider' or 'model'".to_string(),
            ));
        }
    }

    let query = MetricsTimeseriesQuery {
        time_range: to_time_range(params.start, params.end),
        granularity: params.granularity,
        filter: to_dimension_filter(&params.provider, &params.model, &params.server_ip),
        fields: fields.clone(),
        group_by: params.group_by,
    };

    let rows = storage.query_metrics_timeseries(&query).await?;

    // Pivot rows into the response format: shared timestamps + per-field/group series.
    // Collect unique timestamps in order.
    let mut timestamps: Vec<i64> = Vec::new();
    // Map: (field_name, group) -> Vec<Option<f64>>
    let mut series_map: std::collections::BTreeMap<(String, Option<String>), Vec<Option<f64>>> =
        std::collections::BTreeMap::new();

    for row in &rows {
        let ts_idx = if let Some(pos) = timestamps.iter().position(|&t| t == row.timestamp) {
            pos
        } else {
            timestamps.push(row.timestamp);
            // Extend all existing series with None for the new timestamp
            for values in series_map.values_mut() {
                values.push(None);
            }
            timestamps.len() - 1
        };

        for (i, field) in fields.iter().enumerate() {
            let key = (field.clone(), row.group.clone());
            let values = series_map
                .entry(key)
                .or_insert_with(|| vec![None; timestamps.len()]);
            // Ensure the values vec is long enough
            while values.len() < timestamps.len() {
                values.push(None);
            }
            values[ts_idx] = row.values.get(i).copied().flatten();
        }
    }

    // Ensure all series have the same length
    for values in series_map.values_mut() {
        while values.len() < timestamps.len() {
            values.push(None);
        }
    }

    let series: Vec<TimeseriesSeries> = series_map
        .into_iter()
        .map(|((name, group), values)| TimeseriesSeries {
            name,
            group,
            values,
        })
        .collect();

    Ok(ApiResponse::ok(TimeseriesData { timestamps, series }))
}

```rust
use std::sync::Arc;

use axum::extract::{Path, Query, State};
use axum::response::IntoResponse;
use serde::Deserialize;
use serde::Serialize;

use ts_storage::StorageBackend;
use ts_storage::query::CallsQuery;

use crate::params::{parse_csv, to_dimension_filter, to_time_range};
use crate::response::{ApiError, ApiResponse};

#[derive(Debug, Deserialize)]
pub struct CallsParams {
    pub start: i64,
    pub end: i64,
    #[serde(default)]
    pub provider: Option<String>,
    #[serde(default)]
    pub model: Option<String>,
    #[serde(default)]
    pub server_ip: Option<String>,
    #[serde(default)]
    pub status_code: Option<String>,
    #[serde(default)]
    pub finish_reason: Option<String>,
    #[serde(default = "default_sort_by")]
    pub sort_by: String,
    #[serde(default = "default_sort_order")]
    pub sort_order: String,
    #[serde(default = "default_page")]
    pub page: u32,
    #[serde(default = "default_page_size")]
    pub page_size: u32,
}

fn default_sort_by() -> String { "request_time".to_string() }
fn default_sort_order() -> String { "desc".to_string() }
fn default_page() -> u32 { 1 }
fn default_page_size() -> u32 { 50 }

pub async fn list(
    State(storage): State<Arc<dyn StorageBackend>>,
    Query(params): Query<CallsParams>,
) -> Result<impl IntoResponse, ApiError> {
    let page_size = params.page_size.min(200);

    let status_codes: Vec<u16> = parse_csv(&params.status_code)
        .iter()
        .filter_map(|s| s.parse().ok())
        .collect();

    let query = CallsQuery {
        time_range: to_time_range(params.start, params.end),
        filter: to_dimension_filter(&params.provider, &params.model, &params.server_ip),
        status_codes,
        finish_reasons: parse_csv(&params.finish_reason),
        sort_by: params.sort_by,
        sort_order: params.sort_order,
        page: params.page,
        page_size,
    };

    let page = storage.query_calls(&query).await?;
    Ok(ApiResponse::ok(page))
}

pub async fn detail(
    State(storage): State<Arc<dyn StorageBackend>>,
    Path(id): Path<String>,
) -> Result<impl IntoResponse, ApiError> {
    let call = storage.query_call_by_id(&id).await?;
    match call {
        Some(detail) => Ok(ApiResponse::ok(detail)),
        None => Err(ApiError::NotFound(format!("call not found: {id}"))),
    }
}
```
